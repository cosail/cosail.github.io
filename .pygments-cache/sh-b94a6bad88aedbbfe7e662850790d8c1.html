#!/bin/bash

touch file_one
rm -f file_two

#AND列表:&&
#[ -f file_two ]测试失败,echo " two"不会执行
if [ -f file_one ] && echo "one" && [ -f file_two ] && echo " two"
then
    echo "true"
else
    echo "false"
fi


#OR列表:||
rm -f file_one

if [ -f file_one ] || echo "one" || echo "One"
then
    echo "true"
else
    echo "false"
fi


#混用AND和OR列表: 这里没有用if
#如果测试为真执行命令1, 为假执行命令2
[ -f file_one ] && echo "test for true: cmd1" || echo "test for false: cmd2"

exit 0
