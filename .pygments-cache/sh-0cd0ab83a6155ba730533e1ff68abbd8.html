#!/bin/bash

#
# Shell命令
#

#命令1:"break"
echo ----------------break----------------
rm -rf fred*
echo > fred1    #new file
echo > fred2
mkdir fred3     #new directory
echo > fred4

for file in fred*
do
    if [ -d "$file" ]; then
        break;                  #是目录则跳出
    fi
done

echo I got a directory: $file

rm -rf fred*

#命令2:"continue"
echo ---------------continue--------------
rm -rf fred*
echo > fred1    #new file
echo > fred2
mkdir fred3     #new directory
echo > fred4

for file in fred*
do
    if [ -d "$file" ]; then
        continue;                  #是目录则马上进入下一次循环
    fi
    echo I got a file: $file
done

rm -rf fred*

echo ---------------:-------------------
#命令3:":"        //一个空命令,相当于true
if [ -f fred ]; then
    :
else
    echo "file fred did not exist"
fi


echo -------------.---------------------
#命令4:"."      //在当前shell中执行命令,而不是在子shell中执行(正如source命令,会改变当前环境)
echo 请在shell下用". ./old_version_set.sh",". ./new_version_set.sh"来查看效果


exit 0
